oasm/
├─ bindings/          # Language/runtime bindings (Rust ↔ C, Python via PyO3, C FFI)
├─ compiler/          # OASM compiler core logic
├─ docs/              # Documentation, guides, design notes
├─ logs/              # General runtime logs (human-readable)
├─ manifests/         # YAML manifests (user-facing configs)
├─ native_libs/       # DLLs and native libraries
├─ runtime/           # Runtime artifacts + daemon crate
│  ├─ daemon/         # Supervisor-loop daemon crate
│  │  ├─ Cargo.toml
│  │  ├─ src/
│  │  │  ├─ main.rs          # entrypoint, starts supervisor loop
│  │  │  ├─ supervisor.rs    # orchestrator: spawns tasks, manages retries, mode switch
│  │  │  ├─ watch.rs         # watcher task (notify)
│  │  │  ├─ parser.rs        # YAML parser task
│  │  │  ├─ validator.rs     # schema validator task
│  │  │  ├─ handler.rs       # error classification + repair templates
│  │  │  ├─ converter.rs     # CBOR encoder task
│  │  │  ├─ commit.rs        # atomic write + lineage logging
│  │  │  └─ types.rs         # shared structs (AppState, WatchEvent, etc.)
│  │  ├─ templates/          # repair-template JSON files
│  │  ├─ lineage/            # append-only CBOR/JSON lineage logs
│  │  ├─ debug/              # troubleshooting logs (verbose mode)
│  │  ├─ cbor/               # committed CBOR snapshots (runtime.cbor, last_good.cbor)
│  │  └─ capabilities/       # capability manifests, feature toggles
│  ├─ runtime.cbor           # current CBOR snapshot (symlink or copy from daemon/cbor)
│  └─ last_good.cbor         # fallback snapshot for rollback
├─ schemas/           # Schema definitions (oasm.schema.yaml, JSON Schema, etc.)
├─ shells/            # PowerShell / shell integrations
└─ ui/                # Rust UI crate
   ├─ Cargo.toml
   └─ src/
      ├─ main.rs
      ├─ config.rs          # AppState, Lifecycle, helpers
      ├─ startup.rs         # environment init
      └─ other UI modules
